Informa las beans de esquema incorrectas definidas en los contextos de aplicación XML.


  <beans ... xmlns:util="https://www.springframework.org/schema/util">
    <bean class="MiBean" id="miBean"/>
    <bean class="ErrTypeBean" id="errTypeBean"/>
    <util:list list-class="java.lang.String"/> // Informa "La clase debe ser heredera de 'java.util.List'"
    <util:set value-type="MiBean">
      <ref bean="fooBean"/>
      <ref bean="errTypeBean"/> // Informa "No se puede agregar a la colección de tipo 'MiBean'"
    </util:set>
  </beans>
