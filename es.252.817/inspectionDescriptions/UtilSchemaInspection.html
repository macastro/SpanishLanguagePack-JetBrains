Informa sobre beans de esquema definidos incorrectamente en el contexto de la aplicación XML.


  <beans ... xmlns:util="https://www.springframework.org/schema/util">
    <bean class="MyBean" id="myBean"/>
    <bean class="ErrTypeBean" id="errTypeBean"/>
    <util:list list-class="java.lang.String"/> // Informa "la clase debe ser heredera de 'java.util.List'"
    <util:set value-type="MyBean">
      <ref bean="fooBean"/>
      <ref bean="errTypeBean"/> // Informa "no se puede agregar a una colección de tipo 'MyBean'"
    </util:set>
  </beans>
