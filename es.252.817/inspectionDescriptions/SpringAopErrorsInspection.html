Reporta consejos y elementos de asesor incorrectos si no hay atributos pointcut o pointcut-ref.

Ejemplo:


<beans>
    <aop:config>
      <aop:pointcut id="zzz" expression="args()"/>
        <aop:aspect ref="xxx">
            <aop:before method="writeArg1" pointcut="execution(* *(String))"/>
            <aop:after method="aaa" pointcut-ref="zzz"/>
            <aop:after-throwing method="aaa"/> <!-- Se debe definir el atributo 'pointcut' o 'pointcut-ref' -->
       </aop:aspect>
    </aop:config>
</beans>
