Informa sobre llamadas redundantes a la función with que no acceden a ningún contenido del receptor.

Ejemplo:


  class MyClass {
      fun f(): String = ""
  }
  fun testRedundant() {
      with(c) { // <== 'with' es redundante ya que 'c' no se utiliza
          println("1")
      }
  }
  fun testOk() {
      val c = MyClass()
      with(c) { // <== Correcto, ya que 'f()' es un 'c.f()' válido
          println(f())
      }
  }
