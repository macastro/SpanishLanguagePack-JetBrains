Reporta definiciones y usos inválidos de TypedDict.

Ejemplo:


from typing import TypedDict
class Movie(TypedDict):
    name: str
    year: int
    rate: int = 10  # El valor del lado derecho no está soportado
    def method(self): # Las declaraciones en TypedDict no son válidas
        pass
m = Movie(name="name", year=1000, rate=9)
print(m["director"])  # La clave 'director' no existe en 'Movie'
del m["name"]  # No se puede eliminar la clave 'name'
m["year"] = "1001"  # Debería ser 'int', se recibió 'str'
