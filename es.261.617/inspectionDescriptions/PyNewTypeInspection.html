Reporta el uso inválido de NewType.

Ejemplo: 


  from typing import NewType
  InvalidName = NewType("Name", int)  # El nombre de variable 'InvalidName' no coincide con el nombre de NewType 'Name'


  from typing import Literal
  InvalidType = NewType("InvalidType", Literal[1])  # NewType no puede usarse con 'Literal[1]'


  Base = NewType("Base", str)
  class Derived(Base):  # 'Base' cannot be subclassed
      pass
