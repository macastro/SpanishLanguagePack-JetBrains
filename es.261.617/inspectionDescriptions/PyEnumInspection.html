Reporta definiciones y usos inválidos de Enum.

Ejemplo: 


from enum import Enum
class Shape(Enum):
    SQUARE = 1
    CIRCLE = 2
class ExtendedShape(Shape):  # La clase enum 'Shape' es final y no puede ser subclasificada
    TRIANGLE = 3


  from enum import Enum
  class Color(Enum):
    _value_: int
    RED = 1
    GREEN = "green"  # El tipo 'str' no es asignable al tipo declarado 'int'


  from enum import Enum
  class Pet(Enum):
    CAT = 1
    DOG: int = 2  # Los miembros de enum no permiten anotaciones de tipo
