Reporta llamadas redundantes a la funci√≥n with que no acceden a nada del receptor.

Ejemplos:


  class MyClass {
      fun f(): String = ""
  }

  fun testRedundant() {
      with(c) { // <== 'with' es redundante ya que 'c' no se usa
          println("1")
      }
  }

  fun testOk() {
      val c = MyClass()
      with(c) { // <== OK porque 'f()' es efectivamente 'c.f()'
          println(f())
      }
  }
