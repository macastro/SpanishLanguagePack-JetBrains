Reporta métodos @EventListener incorrectos.

Ejemplo:


  @Configuration @ComponentScan
  open class Config
  data class MyEvent(val string: String)
  @Component
  class LogComponent {
    @EventListener // El método con la anotación @EventListener debe ser público
    private fun logCommands(commandName: MyEvent) {}
    @EventListener // El método debe tener como máximo un parámetro
    fun processCommand(commandName: MyEvent, text: String) {}
  }
