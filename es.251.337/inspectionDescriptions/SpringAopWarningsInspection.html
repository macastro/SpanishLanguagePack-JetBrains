Informa sobre aspectos y puntos de corte incorrectos. Cuando algún consejo de AOP coincide con un bean de aspecto, puede provocar errores en tiempo de ejecución.

Ejemplo: 


<beans>
  <aop:config>
    <aop:pointcut expression="args()"/> <!--  El punto de corte debe tener id -->
    <aop:aspect/> <!--  Se debe definir el atributo 'ref' -->
  </aop:config>
</beans>

<beans>
  <bean id="aspect1" class="Aspect1"/>
  <bean id="aspect2" class="Aspect2"/>
  <aop:config>
    <aop:aspect ref="aspect1"> <!--  Algunos avisos coinciden con este bean -->
    <aop:before method="xxx" pointcut="execution(* foo())"/>
    </aop:aspect>
    <aop:aspect ref="aspect2">
      <aop:before method="xxx" pointcut="execution(* foo())"/>
    </aop:aspect>
  </aop:config>
</beans>
