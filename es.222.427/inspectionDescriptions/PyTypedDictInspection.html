Se informa sobre definiciones y usos no válidos de TypedDict.

Ejemplo:


from typing import TypedDict


class Movie(TypedDict):
    name: str
    year: int
    rate: int = 10  # No se admiten valores del lado derecho

    def method(self): # Sentencia no válida en TypedDict
        pass


m = Movie(name="name", year=1000, rate=9)
print(m["director"])  # No existe la clave 'director' en 'Movie'
del m["name"]  # No se puede eliminar la clave 'name'
m["year"] = "1001"  # Se esperaba 'int', pero se obtuvo 'str'
