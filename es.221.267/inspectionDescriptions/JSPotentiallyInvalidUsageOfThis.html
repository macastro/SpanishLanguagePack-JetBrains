Reporta un this en un cierre que se usa para referenciar propiedades de un contexto externo.

Ejemplo:


function Outer() {
  this.outerProp = 1;
  function inner() {
    // malo, porque 'outerProp' de Outer
    // no será actualizado aquí
    // al llamar a 'new Outer()' como se podría esperar
    this.outerProp = 2;
  }
  inner();
}
