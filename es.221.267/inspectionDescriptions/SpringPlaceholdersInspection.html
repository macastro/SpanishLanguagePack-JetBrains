Reporta los placeholders no resueltos configurados en los contextos XML de Spring.

Ejemplo:


  <beans>
    <ctx:property-placeholder location="application.properties"/>

    <bean class="beans.ServerComponent">
      <property name="serverName" value="${server.name}}"/>
      <property name="port" value="${unknown.property}}"/> <!-- reporta "No se puede resolver la clave de la propiedad" -->
    </bean>
  </beans>
